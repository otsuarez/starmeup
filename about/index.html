About

This site was created using Github Pages and mkdocs

setup

pre requirements

sudo pip install mkdocs

commands

git checkout --orphan doc
git rm -fr .
mkdocs new .
cat <<_EOF_ > mkdocs.yml
site_name: Doc
pages:
- [index.md, Home]
- [about.md, About]
theme: united
_EOF_
mkdir docs
cat <<_EOF_ > docs/index.md
# Documentation
_EOF_
cat <<_EOF_ > docs/about.md
# About
_EOF_
mkdocs build
git add .
git commit -am"doc repo update"
git push -u origin doc

git checkout --orphan gh-pages
git rm -rf .
git checkout doc:site/ .
git add .
git commit -am"doc site update"
git push -u origin gh-pages
git checkout master
